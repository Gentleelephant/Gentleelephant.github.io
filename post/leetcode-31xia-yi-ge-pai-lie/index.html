LeetCode: 31.下一个排列

2020-08-12 leetcode

思路：

class Solution {
    public void nextPermutation(int[] nums) {
        if(nums.length == 0 || nums.length == 1){
            return;
        }
        int i = nums.length - 2;
        int j = nums.length - 1;
        int k = nums.length - 1;
        // 查找第一个升序序列
        while(i >= 0){
            if(nums[i] >= nums[j]){
                i--;
                j--;
            }else{
                break;
            }
        }
        // 根据条件进行交换
        if(i >= 0){
            while(nums[i] >= nums[k]){
                k--;
            }
            int t = nums[i];
            nums[i] = nums[k];
            nums[k] = t;
        }
        Arrays.sort(nums,j,nums.length);

    }
}

算法题：给定任意一个自然数，获取它重新排列后，下一个比它大的自然数

public class Problem031 {

    public static void main(String[] args) {

        int n = 1243;
        int length = String.valueOf(n).length();
        int[] nums = new int[length];
        // 把自然数分割成一个一个的数字
        while (n != 0) {
            nums[--length] = n % 10;
            n = n / 10;
        }
        for (int num : nums) {
            System.out.println(num);
        }

        System.out.println("======================");
        method(nums);

        for (int num : nums) {
            System.out.println(num);
        }

    }

    public static void method(int[] nums) {
        if (nums.length == 0 || nums.length == 1) {
            return;
        }
        int i = nums.length - 2;
        int j = nums.length - 1;
        int k = nums.length - 1;
        // 查找第一个连续递增序列（2个数的序列）
        while (i >= 0) {
            if (nums[i] >= nums[j]) {
                i--;
                j--;
            } else {
                break;
            }
        }
        if (i >= 0) {
            while (nums[i] > nums[k]) {
                k--;
            }
            // 进行交换
            int t = nums[i];
            nums[i] = nums[k];
            nums[k] = t;
        }
        // 把后半段进行从小到大排序
        Arrays.sort(nums, j, nums.length);
    }
}
