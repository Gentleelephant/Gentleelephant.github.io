    一、岛屿的周长

    leetcode-463 岛屿的周长

    class Solution {
    public int islandPerimeter(int[][] grid) {
        int sum = 0;
        for(int i = 0;i < grid.length;i++){
            for(int  j = 0;j < grid[0].length;j++){
                if(grid[i][j] == 1){{
                    sum += dfs(grid,i,j);
                }}
            }
        }
        return sum;
    }

    public int dfs(int[][] grid,int r,int c){
        // 遍历到陆地外面（越界）会产生一条边
        if(!isArea(grid,r,c)){
            return 1;
        }
        // 遍历到水域会产生一条边
        if(grid[r][c] == 0){
            return 1;
        }
        // 遍历到已经遍历过的陆地不会产生边
        if(grid[r][c] != 1){
            return 0;
        }
        grid[r][c] = 2;

        return dfs(grid,r-1,c) + dfs(grid,r+1,c) + dfs(grid,r,c-1) + dfs(grid,r,c+1);
    }

    public boolean isArea(int[][] grid,int r,int c){
        return r >= 0 && r < grid.length && c >= 0 && c < grid[0].length;
    }


}

    二、岛屿的最大面积

    leetcode-463 岛屿的最大面积

    class Solution {
    public int maxAreaOfIsland(int[][] grid) {
        int count = 0;
        for(int i = 0;i < grid.length;i++){
            for(int j = 0;j < grid[0].length;j++){
                if(grid[i][j] == 1){
                    int temp = dfs(grid,i,j);
                    count = count > temp ? count : temp;
                }
            }
        }
        return count;
    }

    public int dfs(int[][] grid,int r,int c){
        // 出界
        if(!isArea(grid,r,c)){
            return 0;
        }
        // 遇到水域
        if(grid[r][c] != 1){
            return 0;
        }
        // 设置当前位置已经被访问过
        grid[r][c] = 2;

        return dfs(grid,r-1,c) + dfs(grid,r+1,c) + dfs(grid,r,c - 1) + dfs(grid,r,c + 1) + 1;

    }
    // 判断当前位置是否出界
    public boolean isArea(int[][] grid,int r,int c){
        return r >= 0 && r < grid.length && c >= 0 && c < grid[0].length;
    }

}

    三、统计封闭岛屿的数目

    leetcode-463 岛屿的最大面积

    思路：封闭岛屿说明不能走到边缘

    class Solution {
    public int closedIsland(int[][] grid) {
        int count = 0;
        for(int i = 0;i < grid.length;i++){
            for(int j = 0;j < grid[i].length;j++){
                if(grid[i][j] == 0){
                    if(dfs(grid,i,j)){
                        count++;
                    }
                }
            }
        }
        return count;
    }

    public boolean dfs(int grid[][],int r,int c){
        // 如果接触到边界说明不是岛屿
        if(!isArea(grid,r,c)){
            return false;
        }

        if(grid[r][c] != 0){
            return true;
        }

        grid[r][c] = 2;

        boolean flage1 = dfs(grid,r-1,c);
        boolean flage2 = dfs(grid,r+1,c);
        boolean flage3 = dfs(grid,r,c+1);
        boolean flage4 = dfs(grid,r,c-1);

        return flage1 && flage2 && flage3 && flage4;
    }

    public boolean isArea(int[][] grid,int r,int c){
        return r >= 0 && r < grid.length && c >= 0 && c < grid[0].length;
    }
}
