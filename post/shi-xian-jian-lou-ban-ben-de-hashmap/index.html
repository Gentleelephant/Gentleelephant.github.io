实现简陋版本的HashMap

2020-07-16 集合

非泛型

// HashMap中存储的节点类型
public class Node {

    int hash;
    Object key;
    Object value;
    Node next;
}

// 实现类
public class MyHash01 {

    public static int LENGTH = 16;


    Node[] table;       // 位桶数组
    int size;           // 存放的键值对个数

    public MyHash01() {
        // 长度一般要是2的整数次幂
        table = new Node[LENGTH];
    }

    @Override
    public String toString() {
        StringBuilder sb = new StringBuilder("{");
        for (Node node : table) {
            while (node != null){
                sb.append(node.key + ":" + node.value + ",");
                node = node.next;
            }
        }
        sb.setCharAt(sb.length()-1, '}');
        return sb.toString();
    }


    public Object get(Object key){
        int hash = this.hash(key.hashCode(), table.length);
        if (table[hash] != null){
            Node temp = table[hash];
            while (temp != null){
                if (temp.key.equals(key)){
                    // 如果两个key相等，就查找到了
                    System.out.println("查找到了:"+temp.value);
                    return temp.value;
                }else {
                    temp = temp.next;
                }
            }
        }
        return null;
    }


    public void put(Object key, Object value){

        /**
         * TODO：数组扩容
         */

        Node node = new Node();
        node.hash = this.hash(key.hashCode(), table.length);
        node.key = key;
        node.value = value;
        node.next = null;

        Node temp = table[node.hash];
        Node lastNode = null;
        // 如果此节点为空，直接添加
        if (temp == null){
            table[node.hash] = node;
            size++;
        }else {
            // 如果此节点已经有了值，就遍历对应的链表
            while (temp != null){
                // key重复就覆盖value
                if (temp.key.equals(key)){
                    System.out.println("key已经存在");
                    temp.value = value;
                    return;
                }else {
                    // key不重复
                    // 把最后一个结点的地址保存下来
                    if (temp.next == null){
                        lastNode = temp;
                    }
                    temp = temp.next;
                }
            }
            // 把要添加的节点赋值到链表的最后
            lastNode.next = node;
            size++;
        }
    }

    public int hash(int v,int length){
        System.out.println("hash:" + (v&(length - 1)));
        return v&(length - 1);
    }

    public static void main(String[] args) {

        MyHash01 myHash01 = new MyHash01();

        myHash01.put(53, "aaa");
        myHash01.put(69, "bbb");
        myHash01.put(85, "ccc");
        myHash01.get(85);
        System.out.println("size:" + myHash01.size);
        System.out.println(myHash01.toString());
//        System.out.println("结束");
    }
}
