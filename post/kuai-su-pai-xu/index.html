快速排序

2020-07-30 
private void quickSort(int[] nums, int start, int end){
        if (start >= end) {
            return;
        }
        int i = start;
        int j = end;
        int key = nums[i];
        while (i < j) {
            while (i < j && nums[j] >= key) {
                j--;
            }
            nums[i] = nums[j];
            while (i < j && nums[i] <= key) {
                i++;
            }
            nums[j] = nums[i];
        }
        nums[i] = key;
        quickSort(nums, start, i - 1);
        quickSort(nums, i + 1, end);
    }
